#!/bin/bash
# Log de user data
exec > /var/log/user-data.log 2>&1
set -x

# Actualizar paquetes e instalar Docker
yum update -y
yum install -y docker git

# Habilitar y arrancar Docker
systemctl enable docker
systemctl start docker
usermod -aG docker ec2-user

# Instalar Docker Compose
mkdir -p /usr/local/lib/docker/cli-plugins
curl -SL https://github.com/docker/compose/releases/download/v2.29.2/docker-compose-linux-x86_64 -o /usr/local/lib/docker/cli-plugins/docker-compose
chmod +x /usr/local/lib/docker/cli-plugins/docker-compose

# Crear estructura del proyecto
mkdir -p /home/ec2-user/tienda-tecnologica
cd /home/ec2-user/tienda-tecnologica

# Crear archivo .env con variables de RDS
cat > .env << EOF
DB_HOST=${rds_endpoint}
DB_USER=admin
DB_PASSWORD=${db_password}
DB_NAME=tiendatechdb
DB_PORT=3306
EOF

# Crear docker-compose.yml
cat > docker-compose.yml << 'EOF'
version: '3.8'
services:
  backend:
    image: node:18-alpine
    container_name: tienda-tech-backend
    working_dir: /app
    command: sh -c "npm install && node server.js"
    ports:
      - "3001:3001"
    environment:
      - DB_HOST=${DB_HOST}
      - DB_USER=${DB_USER}
      - DB_PASSWORD=${DB_PASSWORD}
      - DB_NAME=${DB_NAME}
      - DB_PORT=${DB_PORT}
    volumes:
      - ./backend:/app
    restart: always

  frontend:
    image: nginx:alpine
    container_name: tienda-tech-frontend
    ports:
      - "80:80"
    volumes:
      - ./frontend:/usr/share/nginx/html
      - ./frontend/nginx.conf:/etc/nginx/conf.d/default.conf
    restart: always
EOF

# Crear directorios y archivos mínimos para prueba
mkdir -p backend frontend

# Backend simple (server.js)
cat > backend/server.js << 'EOF'
const express = require('express');
const mysql = require('mysql2');
const app = express();
const port = 3001;

// Conexión a RDS
const db = mysql.createConnection({
  host: process.env.DB_HOST,
  user: process.env.DB_USER,
  password: process.env.DB_PASSWORD,
  database: process.env.DB_NAME,
  port: process.env.DB_PORT
});

db.connect(err => {
  if (err) {
    console.error('Error conectando a DB:', err);
  } else {
    console.log('Conectado a RDS MySQL');
  }
});

app.get('/health', (req, res) => {
  res.status(200).json({ status: 'OK', timestamp: new Date() });
});

app.get('/api/productos', (req, res) => {
  db.query('SELECT * FROM productos', (err, results) => {
    if (err) res.status(500).json(err);
    else res.json(results);
  });
});

app.listen(port, () => {
  console.log(`Backend corriendo en puerto ${port}`);
});
EOF

cat > backend/package.json << 'EOF'
{
  "name": "tienda-backend",
  "version": "1.0.0",
  "dependencies": {
    "express": "^4.18.2",
    "mysql2": "^3.6.0"
  }
}
EOF

# Frontend simple
cat > frontend/index.html << 'EOF'
<!DOCTYPE html>
<html>
<head><title>Tienda Tech</title></head>
<body>
<h1>Tienda Tecnológica - HA en AWS</h1>
<p>Backend API: <span id="status">Cargando...</span></p>
<script>
fetch('/api/productos')
  .then(r => r.json())
  .then(data => document.getElementById('status').innerText = 'Conectado ✅')
  .catch(() => document.getElementById('status').innerText = 'Error ❌');
</script>
</body>
</html>
EOF

# Iniciar contenedores
cd /home/ec2-user/tienda-tecnologica
docker compose up -d

# Configurar systemd para reinicio automático
cat > /etc/systemd/system/app-compose.service << EOF
[Unit]
Description=Docker Compose App
Requires=docker.service
After=docker.service

[Service]
Type=oneshot
RemainAfterExit=yes
WorkingDirectory=/home/ec2-user/tienda-tecnologica
ExecStart=/usr/local/lib/docker/cli-plugins/docker-compose up -d
ExecStop=/usr/local/lib/docker/cli-plugins/docker-compose down
User=root

[Install]
WantedBy=multi-user.target
EOF

systemctl enable app-compose.service
